//Program to check whether a number is Armstrong or not

#include <bits/stdc++.h>
using namespace std;

bool isamstrong(int num)
{
    int k=to_string(num).length();
    int n=num, sum=0;
    while(n>0)
    {
        int rem=n%10;
        sum+=pow(rem,k);
        n=n/10;
    }
    return sum==num;
}

int main()
{
    int num;
    cout<<"Enter a number:";
    cin>>num;
    
    bool res=isamstrong(num);
    if(res==true)
       cout<<num<<" is Armstrong";
    else
       cout<<num<<" is not an Armstrong";
    return 0;
}
