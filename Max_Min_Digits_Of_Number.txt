//Program to print max digir and min digit in a Number

#include <bits/stdc++.h>
using namespace std;

void findmaxmin(int n)
{
    int max=0, min=9;
    while(n>0)
    {
        int rem=n%10;
        if(rem>max)
           max=rem;
        if(rem<min)
           min=rem;
        n/=10;
    }
    cout<<"Max digit = "<<max<<endl;
    cout<<"Min digit = "<<min;
}

int main()
{
    int n;
    cout<<"Enter a Number:";
    cin>>n;
    findmaxmin(n);
    return 0;
}
