//Program to check whether the given number is prime or not

#include <bits/stdc++.h>
using namespace std;

bool isprime(int n)
{
    if(n<=1)
      return false;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i==0)
           return false;
    }
    return true;
}

int main()
{
    int n;
    cout<<"Enter any number: ";
    cin>>n;
    
    if(isprime(n)==true)
       cout<<n<<" is a Prime number";
    else
       cout<<n<<" is a composite number";
    return 0;
}
