//Program to find sum of 2 fractions

#include <bits/stdc++.h>
using namespace std;

int gcd(int m, int n)
{
    while(m>0 && n>0)
    {
        if(m>n)
           m%=n;
        else
          n%=m;
    }
    if(m==0)
       return n;
    return m;
}

void fraction_sum(int n1,int d1,int n2, int d2)
{
    int n3,d3;
    
    int lcm=(d1*d2)/gcd(d1,d2);
    
    int newn1=n1*(lcm/d1);
    int newn2=n2*(lcm/d2);
    
    n3=newn1 + newn2;
    d3=lcm;
    
    int common=gcd(n3,d3);
    n3/=common;
    d3/=common;
    cout<<"Sum="<<n3<<"/"<<d3;
}

int main()
{
    int n1,n2,d1,d2;
    cout<<"Enter num and den for 2 numbers:"<<endl;
    cin>>n1>>d1>>n2>>d2;
    fraction_sum(n1,d1,n2,d2);
    return 0;
}
