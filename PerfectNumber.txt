//Program to check whether a number is a perfect number or not
//A perfect number is a number that is equals to sum of its proper divisors excluding itself

#include<bits/stdc++.h>
using namespace std;

bool ispernum(int n)
{
    int sum=0;
    for(int i=1; i*i <=n; i++)
    {
      if(n%i==0)
      {
        if(i*i==n || i==1)
           sum+=i;
        else
           sum=sum+i+(n/i);
      }
    }
    if(sum==n)
        return true;
    return false;
}

int main()
{
    int n;
    cout<<"Enter a number: ";
    cin>>n;
    
    bool res=ispernum(n);
    if(res==true)
       cout<<n<<" is a perfect number";
    else
       cout<<n<<" is not a perfect number";
    return 0;
}
