//Program to print all Factors of a number

#include <bits/stdc++.h>
using namespace std;

void factors(int n)
{
    vector<int>fact;
    for(int i=1;i*i <= n; i++)
    {
        if(n%i==0)
        {
            fact.push_back(i);
            if(i != n/i)
              fact.push_back(n/i);
        }
    }
    sort(fact.begin(),fact.end());
    for(int i:fact)
       cout<<i<<" ";
}

int main()
{
    int n;
    cout<<"Enter a number:";
    cin>>n;
    factors(n);
    return 0;
}
