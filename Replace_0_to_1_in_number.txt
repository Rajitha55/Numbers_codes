//Program to Replace all 0 digits to 1 in a number

#include <bits/stdc++.h>
using namespace std;

int replace(int n)
{
    int rev=0,res=0;
    while(n>0)
    {
        int dig=n%10;
        if(dig==0)
           dig=1;
        rev=(rev*10)+dig;
        n/=10;
    }
    while(rev>0)
    {
        res=(res*10)+(rev%10);
        rev/=10;
    }
    return res;
}

int main()
{
    int n;
    cout<<"Enter a number:"<<endl;
    cin>>n;
    int res=replace(n);
    cout<<res;
    return 0;
}
