//Program to check whether a number is Harshad number or not
//Harshad number is a number whose sum of digits will be divisible by the original number
//n%(sum of digits of n) == 0

#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cout<<"Enter a number:";
    cin>>n;
    int sum=0, temp=n;
    while(temp>0)
    {
        sum+=(temp%10);
        temp/=10;
    }
    if(n%sum == 0)
       cout<<n<<" is a Harshad number";
    else
       cout<<n<<" is not a Harshad number";
       
    return 0;
}
