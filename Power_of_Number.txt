//Program to print power of a number
//x=number, n=power

#include <bits/stdc++.h>
using namespace std;

double power(double x, int n)
{
    double res=1;
    if(x==0 || x==1)
       return x;
       
    if(n<0)  //If the poer(n) is a negative value
    {
        x=1/x;
        n=-(n+1);
        res=res*x;
    }

    while(n>0)
    {
        if(n%2==0)
        {
            n=n/2;
            x=x*x;
        }
        else
        {
            res=res*x;
            n=n-1;
        }
    }        
    
    return res;
}

int main()
{
    double x;
    int n;
    cout<<"Enter number and power you want to print:";
    cin>>x>>n;
    double res=power(x,n);
    cout<<x<<" power of "<<n<<"= "<<res;
    return 0;
}
