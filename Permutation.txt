//Program to print Permutation

#include <bits/stdc++.h>
using namespace std;

int permutation(int n, int r)
{
    int res=1;
    for(int i=n; i>= n-r+1; i--)
       res*=i;
    return res;
}

int main()
{
    int n,r;
    cout<<"Print values of n and r:";
    cin>>n>>r;
    int res=permutation(n,r);
    cout<<res;
    return 0;
}
