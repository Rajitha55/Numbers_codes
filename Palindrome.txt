//Program to check whether the given number is a palindrome or not

#include <bits/stdc++.h>
using namespace std;

bool palindrome(int num)
{
    int dup=num, rev=0;
    while(dup>0)
    {
        rev=(rev*10)+(dup%10);
        dup/=10;
    }
    cout<<"Reverse of given number is: "<<rev<<endl;;
    if(rev==num)
        return true;
    return false;
}

int main()
{
    int num;
    cout<<"Enter a number to check whether its a palindrome or not:";
    cin>>num;
    bool res=palindrome(num);
    if(res==true)
       cout<<"Heknce, Entered number is a Palindrome";
    else
       cout<<"Hence, Entred number is not a palindrome";
    return 0;
}
