//Program to print sum of natural numbers

#include<bits/stdc++.h>
using namespace std;

int sum(int n)
{
    return (n*(n+1))/2;
}

int main()
{
    int n;
    cout<<"Enter a number: ";
    cin>>n;
    int res=sum(n);
    cout<<"Sum of natural numbers till "<<n<<" ="<<res;
    return 0;
}
